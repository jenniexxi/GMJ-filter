const getUrlParams = () => {
  let params = {};
  window.location.search.replace(
    /[?&]+([^=&]+)=([^&]*)/gi,
    function (str, key, value) {
      params[key] = decodeURIComponent(value);
    }
  );
  return params;
};


 urlParamsObj = getUrlParams();
